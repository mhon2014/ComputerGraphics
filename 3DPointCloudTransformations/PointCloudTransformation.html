def homogenous(array):
    return np.vstack((array, np.ones([1, array.shape[1]])))
                                    
def translation(tx, ty, tz):
    return np.array([[1, 0, 0, tx],
                     [0, 1, 0, ty],
                     [0, 0, 1, tz]])

def scaling(sx, sy, sz):
    return np.array([[sx, 0, 0],
                     [0, sy, 0],
                     [0, 0, sz]])
        
def roll(theta):
    return np.array([[1, 0, 0],
                     [0, cos(d2r(theta)), -sin(d2r(theta))],
                     [0, sin(d2r(theta)), cos(d2r(theta))]])
    
def pitch(theta):
    return np.array([[cos(d2r(theta)), 0, sin(d2r(theta))],
                     [0, 1, 0],
                     [-sin(d2r(theta)), 0,  cos(d2r(theta))]])

def yaw(theta):
    return np.array([[cos(d2r(theta)), -sin(d2r(theta)), 0],
                     [sin(d2r(theta)), cos(d2r(theta)), 0],
                     [0, 0, 1]])


'''  Rotation function for all axis  '''
def rotation(xdeg, ydeg, zdeg):
    return yaw(zdeg)@pitch(ydeg)@roll(xdeg)
